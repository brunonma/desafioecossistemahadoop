def clean_text(text,stop_word = ''):
  #print(stop_word)
  ## minsculo
    text = text.lower()

  ## limpar os nmero dos jogadores (numeroNomejogado)
    text = re.sub('\d+\-'," ",text) #re.sub('[\d+]\w',"*",dados.texto_limpo[3])

    text = re.sub('||'," ",text) #re.sub('[\d+]\w',"*",dados.texto_limpo[3])

    text = remove_punctuation(text) ## remover pontuaes
    text = remove_stopwords(text,stop_word) ## remover stop words
    return text
